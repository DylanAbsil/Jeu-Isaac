161  Kr_Log_Print(KR_LOG_INFO, "Level : %s has been loaded !\n", pLevel->szLevelFile);
162  return TRUE;

196 Boolean Kr_Level_Layout(Kr_Level *pLevel, FILE *pFile)
197 {
198  Sint32 i, j, iTmp;
199 
200  fscanf(pFile, "%d %d", &pLevel->iLevel_TileWidth, &pLevel->iLevel_TileHeight);
201 
202  /* Allocation du tableau 2D szLayout */
203  pLevel->szLayout = malloc(pLevel->iLevel_TileWidth*sizeof(Uint32*));
204  if (pLevel->szLayout == NULL)
205  {
206  Kr_Log_Print(KR_LOG_ERROR, "Can't allocate the layout of the level \n");
207  return FALSE;
208  }
209  for (i = 0; i < pLevel->iLevel_TileWidth; i++)
210  {
211  pLevel->szLayout[i] = malloc(pLevel->iLevel_TileHeight*sizeof(Uint32));
212  if (pLevel->szLayout[i] == NULL)
213  {
214  Kr_Log_Print(KR_LOG_ERROR, "Can't allocate the layout[i] of the level \n");
215  return FALSE;
216  }
217  }
218  /* Affectation des données level au schema */
219  for (j = 0; j<pLevel->iLevel_TileHeight; j++)
220  {
221  for (i = 0; i<pLevel->iLevel_TileWidth; i++)
222  {
223  fscanf(pFile, "%d", &iTmp);
224  if (iTmp >= pLevel->pLevel_Tileset->iNbTilesX * pLevel->pLevel_Tileset->iNbTilesY) // On vérifie que le codage du tile existe
225  {
226  Kr_Log_Print(KR_LOG_ERROR, "Can't load a tile in the layout because it's out of the tileset file ");
227  Kr_Log_Print(KR_LOG_ERROR, ">> Tile n°\"%d\", but iNbTilesX = %d and iNbTilesY = %d", iTmp, pLevel->pLevel_Tileset->iNbTilesX, pLevel->pLevel_Tileset->iNbTilesY);
228  return FALSE;
229  }
230  pLevel->szLayout[i][j] = iTmp;
231  }
232  }
233  Kr_Log_Print(KR_LOG_INFO, "Kr_Level_Layout: Done\n");
234  return TRUE;
235 }
