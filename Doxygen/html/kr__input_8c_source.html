36  {

43  case SDL_KEYUP:

54  break;
55  case SDL_MOUSEBUTTONUP:

57  break;
58  case SDL_QUIT:
59  inEvent->bQuit = 1;
60  break;
61  case SDL_MOUSEWHEEL:
